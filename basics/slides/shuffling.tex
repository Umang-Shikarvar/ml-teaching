	\begin{frame}{Why shuffle for SGD?}
		\only<1->
		{
			We can fall into a loop!\\
			\vspace{1cm}
			SGD on point 1 : $\theta_0 + 0.2, \theta_1 - 0.2$\\
			SGD on point 2 : $\theta_0 - 0.2, \theta_1 + 0.2$\\
			\vspace{1cm}
			Biased learning as point 2 follows point 1.
		}
